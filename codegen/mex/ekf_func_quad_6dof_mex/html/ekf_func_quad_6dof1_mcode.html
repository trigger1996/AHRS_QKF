1function [X, P]=ekf_func_quad_6dof(X, P, acc, gyro, mag, dt, Q, R)
 2
 3    q0 = X(1);    q1 = X(2);    q2 = X(3);    q3 = X(4);
 4	wx = gyro(1); wy = gyro(2); wz = gyro(3);
 5	bwx = X(5);   bwy = X(6);   bwz = X(7);					% gyro bias
 6	
 7	% 传感器模型
 8	Z = [acc, mag];
 9	
10	% 系统观测模型
11    X = [  q0 - (q1*(wx-bwx)*dt)/2 - (q2*(wy - bwy)*dt)/2 - (q3*(wz - bwz)*dt)/2;
12		   q1 + (q0*(wx-bwx)*dt)/2 - (q3*(wy - bwy)*dt)/2 + (q2*(wz - bwz)*dt)/2;
13		   q2 + (q3*(wx-bwx)*dt)/2 + (q0*(wy - bwy)*dt)/2 - (q1*(wz - bwz)*dt)/2;
14		   q3 - (q2*(wx-bwx)*dt)/2 + (q1*(wy - bwy)*dt)/2 + (q0*(wz - bwz)*dt)/2;
15                                                                             bwx;
16                                                                             bwy;
17                                                                             bwz];
18    % 四元数归一化
19	X(1:4) = X(1:4) / norm(X(1:4));
20	
21	% 状态转移矩阵
22	F = [  0,       - wx/2,       - wy/2,       - wz/2,  q1/2,  q2/2,  q3/2;
23        wx/2,            0,         wz/2,       - wy/2, -q0/2,  q3/2, -q2/2;
24        wy/2,       - wz/2,            0,         wx/2, -q3/2, -q0/2,  q1/2;
25        wz/2,         wy/2,       - wx/2,            0,  q2/2, -q1/2, -q0/2;
26           0,            0,            0,            0,     0,     0,     0;
27           0,            0,            0,            0,     0,     0,     0;
28           0,            0,            0,            0,     0,     0,     0];
29
30	% 四元数更新   
31	q0 = X(1); q1 = X(2); q2 = X(3); q3 = X(4);
32	
33	% 观测协方差更新
34	PHI = eye(7) + F * dt;
35    P = PHI * P * PHI' + Q;
36	
37%%%% 这一坨看不懂	
38	quat = X(1:4);
39	cbn  = my_quat2cbn(quat);
40	mR = cbn*mag';
41    bx = norm([mR(1),mR(2)]);
42    bz = mR(3);
43	
44	h = [cbn'*[0;0;-1];cbn'*[bx;0;bz]];%%计算预测值
45	innov = h - Z'; 
46    
47    H = my_calcH(bx,bz,q0,q1,q2,q3);
48
49    Hk_1 = [  2*q2, -2*q3, 2*q0, -2*q1, 0, 0, 0;
50             -2*q1, -2*q0,-2*q3, -2*q2, 0, 0, 0;
51             -2*q0,  2*q1, 2*q2, -2*q3, 0, 0, 0];
52
53    Hk_2 = 2*[ -2*bz*q2,            2*bz*q3,            -4*bx*q2-2*bz*q0,      -4*bx*q3+2*bz*q1 0 0 0;
54               -2*bx*q3+2*bz*q1,	2*bx*q2+2*bz*q0,	2*bx*q1+2*bz*q3,       -2*bx*q0+2*bz*q2 0 0 0;
55                2*bx*q2,            2*bx*q3-4*bz*q1,	2*bx*q0-4*bz*q2,       2*bx*q1  0 0 0];
56  
57    H = [Hk_1;Hk_2];
58	
59	% 更新卡尔曼增益
60	S = H*P*H' + R;
61    K = P*H'/S;
62	
63	% 卡尔曼更新阶段，公式3-5
64	X = X - K * innov;			%%得到估计状态
65    I = eye(length(P));
66    P = (I - K * H)*P;          % 更新协方差矩阵
67
68	% 归一化四元数
69	X(1:4) = X(1:4)/norm(X(1:4));
70	
71end
